    def _missing_(cls, value: 'int') -> 'ErrorCode':
        """Lookup function used when value is not found.

        if not (isinstance(value, int) and 0x00000000 <= value <= 0xFFFFFFFF):

        if 0x0000000E <= value <= 0xFFFFFFFF:
